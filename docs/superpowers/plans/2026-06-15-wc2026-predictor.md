```js
// 순수 파생/포맷 헬퍼. America/Chicago 고정.
const TZ = 'America/Chicago';

export function verdictFromProb(prob) {
  const win = Number(prob?.win) || 0;
  const draw = Number(prob?.draw) || 0;
  const loss = Number(prob?.loss) || 0;
  const max = Math.max(win, draw, loss);
  if (max === win) return 'home_win';
  if (max === loss) return 'away_win';
  return 'draw';
}

export function confidenceLevel(prob) {
  const max = Math.max(Number(prob?.win) || 0, Number(prob?.draw) || 0, Number(prob?.loss) || 0);
  if (max >= 60) return 'high';
  if (max >= 45) return 'med';
  return 'toss';
}

export function isStale(generatedAtIso, now = new Date(), thresholdHours = 28) {
  const gen = new Date(generatedAtIso).getTime();
  if (Number.isNaN(gen)) return false;
  return (now.getTime() - gen) / 3_600_000 > thresholdHours;
}

function ctParts(date) {
  const s = new Intl.DateTimeFormat('en-CA', { timeZone: TZ, year: 'numeric', month: '2-digit', day: '2-digit' }).format(date);
  const [y, m, d] = s.split('-').map(Number);
  return { y, m, d };
}

function dayDiffCT(from, to) {
  const a = ctParts(from), b = ctParts(to);
  return Math.round((Date.UTC(b.y, b.m - 1, b.d) - Date.UTC(a.y, a.m - 1, a.d)) / 86_400_000);
}

export function formatKickoff(kickoffIso, now = new Date()) {
  const d = new Date(kickoffIso);
  const p = ctParts(d);
  const time = new Intl.DateTimeFormat('en-GB', { timeZone: TZ, hour: '2-digit', minute: '2-digit', hour12: false }).format(d);
  const diff = dayDiffCT(now, d);
  const rel = diff === 0 ? '오늘' : diff === 1 ? '내일' : diff === -1 ? '어제' : `${p.m}/${p.d}`;
  return { time, rel, label: `${rel} ${time} CT` };
}
```

```js
import { verdictFromProb } from './format.js';

export function normalizeProb(prob = {}) {
  const win = Math.max(0, Number(prob.win) || 0);
  const draw = Math.max(0, Number(prob.draw) || 0);
  const loss = Math.max(0, Number(prob.loss) || 0);
  const sum = win + draw + loss;
  if (sum === 0) return { win: 34, draw: 33, loss: 33, normalized: true };
  if (sum === 100) return { win, draw, loss, normalized: false };
  const scaled = {
    win: Math.round((win * 100) / sum),
    draw: Math.round((draw * 100) / sum),
    loss: Math.round((loss * 100) / sum),
  };
  const drift = 100 - (scaled.win + scaled.draw + scaled.loss);
  const key = scaled.win >= scaled.draw && scaled.win >= scaled.loss
    ? 'win' : scaled.loss >= scaled.draw ? 'loss' : 'draw';
  scaled[key] += drift;
  return { ...scaled, normalized: true };
}

const REQUIRED = ['id', 'group', 'matchday', 'home', 'away', 'prob', 'status'];

export function validateData(predictions, teams) {
  const errors = [];
  const warnings = [];
  if (!predictions || !Array.isArray(predictions.matches)) {
    errors.push('predictions.matches가 배열이 아님');
    return { ok: false, errors, warnings };
  }
  const ids = new Set((teams || []).map(t => t.id));
  for (const m of predictions.matches) {
    const where = `match ${m.id ?? '(id없음)'}`;
    for (const f of REQUIRED) {
      if (m[f] === undefined) errors.push(`${where}: 필수 필드 누락 '${f}'`);
    }
    if (m.prob) {
      const s = (Number(m.prob.win) || 0) + (Number(m.prob.draw) || 0) + (Number(m.prob.loss) || 0);
      if (s !== 100) warnings.push(`${where}: prob 합이 100이 아님(${s})`);
      if (m.verdict && m.verdict !== verdictFromProb(m.prob)) {
        warnings.push(`${where}: verdict(${m.verdict})가 prob 최고값과 불일치`);
      }
    }
    if (ids.size) {
      if (m.home && !ids.has(m.home)) warnings.push(`${where}: home 팀 '${m.home}' teams에 없음`);
      if (m.away && !ids.has(m.away)) warnings.push(`${where}: away 팀 '${m.away}' teams에 없음`);
    }
  }
  return { ok: errors.length === 0, errors, warnings };
}
```

```js
export function filterMatches(matches, criteria = {}) {
  const { group = 'ALL', matchday = 'ALL', remainingOnly = true, query = '', teams = [] } = criteria;
  const byId = new Map((teams || []).map(t => [t.id, t]));
  const q = String(query).trim().toLowerCase();
  return (matches || []).filter(m => {
    if (remainingOnly && m.status !== 'upcoming') return false;
    if (group !== 'ALL' && m.group !== group) return false;
    if (matchday !== 'ALL' && String(m.matchday) !== String(matchday)) return false;
    if (q) {
      const h = byId.get(m.home);
      const a = byId.get(m.away);
      const hay = [m.home, m.away, h?.name, a?.name].filter(Boolean).join(' ').toLowerCase();
      if (!hay.includes(q)) return false;
    }
    return true;
  });
}
```

```js
import { formatKickoff, confidenceLevel } from './format.js';

export function esc(s) {
  return String(s ?? '').replace(/[&<>"']/g, c => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[c]));
}

export function renderProbBar(prob) {
  return `<div class="probbar" role="img" aria-label="승 ${prob.win}% 무 ${prob.draw}% 패 ${prob.loss}%">` +
    `<span class="seg seg-home" style="width:${prob.win}%">${prob.win}%</span>` +
    `<span class="seg seg-draw" style="width:${prob.draw}%">${prob.draw}%</span>` +
    `<span class="seg seg-away" style="width:${prob.loss}%">${prob.loss}%</span>` +
    `</div>`;
}

export function verdictLabel(match, byId) {
  const p = match.prob;
  if (match.verdict === 'draw') return `예측: 무승부 (${p.draw}%)`;
  const id = match.verdict === 'home_win' ? match.home : match.away;
  const pct = match.verdict === 'home_win' ? p.win : p.loss;
  const name = byId.get(id)?.name ?? id;
  return `예측: ${name} 승 (${pct}%)`;
}

function renderChips(items = []) {
  return items.map(t => `<span class="chip">${esc(t)}</span>`).join('');
}

export function renderCard(match, byId, now = new Date()) {
  const homeName = esc(byId.get(match.home)?.name ?? match.home);
  const awayName = esc(byId.get(match.away)?.name ?? match.away);
  const k = formatKickoff(match.kickoff, now);
  const conf = confidenceLevel(match.prob);
  return `<article class="card conf-${conf}" data-group="${esc(match.group)}" data-md="${esc(match.matchday)}">` +
    `<div class="card-meta">${esc(match.group)}조 · MD${esc(match.matchday)} · ${esc(k.label)}</div>` +
    `<div class="card-teams"><span class="team home">${homeName}</span><span class="vs">vs</span><span class="team away">${awayName}</span></div>` +
    renderProbBar(match.prob) +
    `<div class="verdict">${esc(verdictLabel(match, byId))}</div>` +
    `<details class="card-more"><summary>분석 보기</summary>` +
      `<p class="rationale">${esc(match.rationale ?? '')}</p>` +
      `<ul class="notes"><li><b>홈</b> ${esc(match.team_notes?.home ?? '')}</li><li><b>원정</b> ${esc(match.team_notes?.away ?? '')}</li></ul>` +
      `<div class="chips">${renderChips(match.key_variables)}</div>` +
      (match.flip_condition && match.flip_condition !== '—' ? `<p class="flip">⚠️ ${esc(match.flip_condition)}</p>` : '') +
      (match.qualification_context ? `<p class="qual">${esc(match.qualification_context)}</p>` : '') +
    `</details>` +
  `</article>`;
}

export function renderEmptyState(msg = '표시할 경기가 없습니다.') {
  return `<div class="empty">${esc(msg)}</div>`;
}

export function renderList(matches, teams, now = new Date()) {
  if (!matches || !matches.length) return renderEmptyState();
  const byId = new Map((teams || []).map(t => [t.id, t]));
  return matches.map(m => renderCard(m, byId, now)).join('\n');
}

export function renderHeader(meta, { stale = false } = {}) {
  const t = meta?.generated_at ? new Date(meta.generated_at) : null;
  const when = t && !Number.isNaN(t.getTime())
    ? new Intl.DateTimeFormat('en-GB', { timeZone: 'America/Chicago', month: 'numeric', day: 'numeric', hour: '2-digit', minute: '2-digit', hour12: false }).format(t)
    : '—';
  return `<h1>WC2026 잔여 조별리그 예측</h1>` +
    `<div class="updated">마지막 업데이트: ${esc(when)} CT ${stale ? '<span class="badge-stale">⚠️ 업데이트 지연</span>' : ''}</div>`;
}
```

```js
import { validateData, normalizeProb } from './validate.js';
import { isStale } from './format.js';
import { filterMatches } from './filters.js';
import { renderList, renderHeader } from './render.js';

const PRED_URL = './data/predictions.json';
const TEAMS_URL = './data/teams.json';
const CACHE_KEY = 'wc2026.cache.v1';

const state = {
  predictions: { matches: [] },
  teams: [],
  criteria: { group: 'ALL', matchday: 'ALL', remainingOnly: true, query: '' },
};

async function fetchJson(url) {
  const r = await fetch(url, { cache: 'no-store' });
  if (!r.ok) throw new Error(`${url}: ${r.status}`);
  return r.json();
}

function banner(msg) {
  const el = document.getElementById('banner');
  if (!el) return;
  el.textContent = msg;
  el.hidden = !msg;
}

async function load() {
  try {
    const [predictions, teams] = await Promise.all([fetchJson(PRED_URL), fetchJson(TEAMS_URL)]);
    localStorage.setItem(CACHE_KEY, JSON.stringify({ predictions, teams }));
    banner('');
    return { predictions, teams };
  } catch (e) {
    const cached = localStorage.getItem(CACHE_KEY);
    if (cached) { banner('데이터를 불러오지 못해 마지막 캐시를 표시합니다.'); return JSON.parse(cached); }
    banner('데이터를 불러올 수 없습니다.');
    return { predictions: { matches: [] }, teams: [] };
  }
}

function normalizeAll(predictions) {
  for (const m of predictions.matches || []) {
    if (m.prob) { const n = normalizeProb(m.prob); m.prob = { win: n.win, draw: n.draw, loss: n.loss }; }
  }
  return predictions;
}

function buildFilterOptions() {
  const ms = state.predictions.matches || [];
  const groups = [...new Set(ms.map(m => m.group))].sort();
  const mds = [...new Set(ms.map(m => String(m.matchday)))].sort();
  document.getElementById('f-group').innerHTML =
    `<option value="ALL">전체 조</option>` + groups.map(g => `<option value="${g}">${g}조</option>`).join('');
  document.getElementById('f-md').innerHTML =
    `<option value="ALL">전체 MD</option>` + mds.map(m => `<option value="${m}">MD${m}</option>`).join('');
}

function renderAll() {
  const now = new Date();
  const stale = state.predictions.generated_at ? isStale(state.predictions.generated_at, now) : false;
  document.getElementById('header').innerHTML = renderHeader(state.predictions, { stale });
  const matches = filterMatches(state.predictions.matches || [], { ...state.criteria, teams: state.teams });
  document.getElementById('list').innerHTML = renderList(matches, state.teams, now);
}

function wireControls() {
  document.getElementById('f-group').addEventListener('change', e => { state.criteria.group = e.target.value; renderAll(); });
  document.getElementById('f-md').addEventListener('change', e => { state.criteria.matchday = e.target.value; renderAll(); });
  document.getElementById('f-remaining').addEventListener('change', e => { state.criteria.remainingOnly = e.target.checked; renderAll(); });
  document.getElementById('f-q').addEventListener('input', e => { state.criteria.query = e.target.value; renderAll(); });
}

async function main() {
  const { predictions, teams } = await load();
  state.predictions = normalizeAll(predictions || { matches: [] });
  state.teams = teams || [];
  const res = validateData(state.predictions, state.teams);
  res.warnings.forEach(w => console.warn('[data]', w));
  res.errors.forEach(er => console.error('[data]', er));
  buildFilterOptions();
  wireControls();
  renderAll();
}

main();
```

````markdown
# WC2026 예측 갱신 루틴

너는 FIFA World Cup 2026 조별리그 분석가다. 매 실행마다 잔여 조별리그 경기의

## 작업 디렉터리
이 저장소(wc2026-predictor)의 루트. 데이터 계약은 `schema/*.schema.json` 및

## 단계
1. **수집(웹서치)**: 현재까지의 조별리그 결과와 각 조 순위, 그리고 각 팀의 최신
````
